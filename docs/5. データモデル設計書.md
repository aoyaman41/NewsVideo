# 5. データモデル設計書

## 5.1 概要

本ドキュメントでは、NewsVideoアプリケーションで使用するデータモデルの詳細を定義する。

---

## 5.2 ファイル構成

### 5.2.1 プロジェクトディレクトリ構造

```
{project_name}.newsproj/
├── project.json          # プロジェクトメタデータ
├── article.json          # 記事データ
├── parts.json            # パート・スクリプトデータ
├── prompts.json          # 画像生成プロンプトデータ
├── images/               # 画像ファイル
│   ├── generated/        # AI生成画像
│   │   ├── panel_001.png
│   │   ├── panel_002.png
│   │   └── thumbnail.png
│   └── imported/         # インポート画像
│       ├── img_001.jpg
│       └── img_002.png
├── audio/                # 音声ファイル
│   ├── part_001.mp3
│   ├── part_002.mp3
│   └── ...
└── output/               # 出力ファイル
    ├── preview/          # プレビュー動画
    └── final/            # 最終出力動画
```

## 5.3 TypeScript型定義

### 5.3.1 プロジェクト (Project)

```typescript
// src/types/project.ts

/**
 * プロジェクトのメタ情報（一覧表示用）
 */
interface ProjectMeta {
  id: string;
  name: string;
  createdAt: string; // ISO 8601
  updatedAt: string; // ISO 8601
  thumbnailPath?: string;
  articleTitle?: string;
  partCount: number;
  status: ProjectStatus;
}

/**
 * プロジェクトのステータス
 */
type ProjectStatus =
  | 'draft'           // 作成中
  | 'script_ready'    // スクリプト生成済み
  | 'images_ready'    // 画像生成済み
  | 'audio_ready'     // 音声生成済み
  | 'complete';       // 動画出力済み

/**
 * プロジェクト全体
 */
interface Project {
  id: string;
  name: string;
  createdAt: string;
  updatedAt: string;
  version: string; // ファイルフォーマットバージョン "1.0.0"

  article: Article;
  parts: Part[];
  imagePrompts: ImagePrompt[];
  images: ImageAsset[];
  thumbnail?: ImageAssetRef;
  settings: ProjectSettings;

  status: ProjectStatus;
}

/**
 * プロジェクト設定
 */
interface ProjectSettings {
  // 動画設定
  videoResolution: VideoResolution;
  fps: 30 | 60;
  videoBitrate: string; // "8M", "12M", etc.
  audioBitrate: string; // "192k", "320k", etc.

  // オープニング/エンディング
  openingVideoPath?: string;
  endingVideoPath?: string;

  // 音声設定
  voiceProfile: string;
  speakingRate: number; // 0.5-2.0
  pitch: number; // -20.0-20.0

  // 画像設定
  imageStylePreset: string;
  defaultAspectRatio: AspectRatio;

  // その他
  autoSaveEnabled: boolean;
  autoSaveIntervalSec: number;
}

type VideoResolution = '1920x1080' | '1280x720' | '3840x2160';
type AspectRatio = '16:9' | '1:1' | '9:16';  // ※4:3は非対応（統一規約）
```

```typescript
// src/types/part.ts

/**
 * パート（動画の1セクション）
 */
interface Part {
  id: string;
  index: number; // 表示順序 (0-based)

  // スクリプト情報
  title: string;
  summary: string;
  scriptText: string; // 読み上げ原稿
  durationEstimateSec: number; // 推定読み上げ時間

  // 音声
  audio?: AudioAsset;

  // 画像割り当て
  panelImages: ImageAssetRef[];

  // コメント（再生成用）
  comments: Comment[];

  // メタ情報
  createdAt: string;
  updatedAt: string;
  scriptGeneratedAt?: string;
  scriptModifiedByUser: boolean;
}

/**
 * コメント（フィードバック）
 */
interface Comment {
  id: string;
  targetType: CommentTargetType;
  text: string;
  createdAt: string;
  handled: boolean;
  handledAt?: string;
  result?: string; // 適用結果の概要
}

type CommentTargetType =
  | 'script'      // スクリプトへのコメント
  | 'image'       // 画像へのコメント
  | 'thumbnail';  // サムネイルへのコメント
```

### 5.3.4 画像 (Image)

```typescript
// src/types/image.ts

/**
 * 画像生成用プロンプト
 * ※スキーマバージョン: v1.1（統一規約準拠）
 * ※`style`ではなく`stylePreset`を使用
 */
interface ImagePrompt {
  id: string;
  type: ImagePromptType;
  partId?: string; // PartPanelの場合

  // プロンプト内容
  prompt: string;
  negativePrompt?: string;
  stylePreset: string;        // ※必須化（v1.1）
  aspectRatio: AspectRatio;
  version: number;            // プロンプトバージョン（履歴管理用）

  // 追加パラメータ
  composition?: string; // 構図の指示
  colorTone?: string; // 色調の指示
  includeText: boolean; // テキストを含めるか

  // メタ情報
  createdAt: string;
  updatedAt: string;
  generatedBy: 'ai' | 'user';
  modifiedByUser: boolean;
}

type ImagePromptType =
  | 'part_panel'    // パート用パネル画像
  | 'thumbnail';    // サムネイル

/**
 * 画像アセット
 */
interface ImageAsset {
  id: string;
  filePath: string; // プロジェクト相対パス
  sourceType: ImageSourceType;
  metadata: ImageMetadata;
}

type ImageSourceType =
  | 'generated'   // AI生成
  | 'imported';   // インポート

/**
 * 画像メタデータ
 */
interface ImageMetadata {
  width: number;
  height: number;
  mimeType: string;
  fileSize: number; // bytes
  createdAt: string;
  promptId?: string; // 生成に使用したプロンプトID
  tags: string[];
}

/**
 * 画像参照（パートへの割り当て用）
 */
interface ImageAssetRef {
  imageId: string;
  order: number; // パート内の表示順序 (0-based)
  displayDurationSec?: number; // 表示時間（指定しない場合は自動計算）
}

/**
 * サムネイル設定
 */
interface ThumbnailSettings {
  imageId?: string; // 選択された画像ID
  textOverlay?: ThumbnailTextOverlay;
}

interface ThumbnailTextOverlay {
  text: string;
  position: 'top' | 'center' | 'bottom';
  fontSize: number;
  fontColor: string;
  backgroundColor?: string;
}
```

### 5.3.5 音声 (Audio)

```typescript
// src/types/audio.ts

/**
 * 音声アセット
 * ※スキーマバージョン: v1.1（統一規約準拠）
 */
interface AudioAsset {
  id: string;
  filePath: string;           // プロジェクト相対パス
  durationSec: number;        // 実測音声長（ffprobeで取得）
  ttsEngine: TTSEngine;       // 使用TTSエンジン（必須）
  voiceId: string;            // 音声ID
  settings: AudioSettings;    // 詳細設定
  generatedAt: string;        // 生成日時（ISO 8601）
}

/**
 * TTSエンジン
 */
type TTSEngine = 'google_tts' | 'macos_tts';

/**
 * 音声生成設定
 */
interface AudioSettings {
  speakingRate: number;       // 話速 0.5-2.0
  pitch: number;              // ピッチ -20.0-20.0
  languageCode: string;       // "ja-JP"
  volumeGainDb?: number;      // ボリューム調整
}

type AudioEncoding = 'mp3' | 'wav' | 'ogg';

/**
 * TTS設定
 */
interface TTSSettings {
  voiceName: string;
  languageCode: string; // "ja-JP"
  speakingRate: number; // 0.25-4.0
  pitch: number; // -20.0-20.0
  volumeGainDb: number; // -96.0-16.0
}

/**
 * 利用可能な音声情報
 */
interface VoiceInfo {
  name: string;
  displayName: string;
  languageCodes: string[];
  gender: 'MALE' | 'FEMALE' | 'NEUTRAL';
  sampleRateHertz: number;
  voiceType: 'STANDARD' | 'WAVENET' | 'NEURAL2' | 'CHIRP3';
}
```

## 5.6 マイグレーション

### 5.6.1 バージョン管理

プロジェクトファイルにはバージョン番号を含め、将来のフォーマット変更に対応する。

```typescript
interface ProjectMigration {
  fromVersion: string;
  toVersion: string;
  migrate: (data: any) => any;
}

const migrations: ProjectMigration[] = [
  {
    fromVersion: '1.0.0',
    toVersion: '1.1.0',
    migrate: (data) => {
      // v1.0 → v1.1 マイグレーション
      // - Part: createdAt, updatedAt, scriptGeneratedAt, scriptModifiedByUser 追加
      // - AudioAsset: ttsEngine, voiceId, settings, generatedAt 追加
      // - ImagePrompt: stylePreset必須化、version追加
      // - ImageAsset.metadata: fileSize追加
      const now = new Date().toISOString();

      return {
        ...data,
        version: '1.1.0',
        parts: data.parts?.map((part: any) => ({
          ...part,
          createdAt: part.createdAt ?? now,
          updatedAt: part.updatedAt ?? now,
          scriptGeneratedAt: part.scriptGeneratedAt ?? now,
          scriptModifiedByUser: part.scriptModifiedByUser ?? false,
          audio: part.audio ? {
            ...part.audio,
            ttsEngine: part.audio.ttsEngine ?? 'google_tts',
            voiceId: part.audio.voiceId ?? part.audio.voiceProfile ?? '',
            settings: part.audio.settings ?? {
              speakingRate: part.audio.speakingRate ?? 1.0,
              pitch: part.audio.pitch ?? 0,
              languageCode: 'ja-JP',
            },
            generatedAt: part.audio.generatedAt ?? part.audio.createdAt ?? now,
          } : undefined,
        })),
        imagePrompts: data.imagePrompts?.map((prompt: any) => ({
          ...prompt,
          stylePreset: prompt.stylePreset ?? prompt.style ?? 'news_panel',
          version: prompt.version ?? 1,
        })),
        images: data.images?.map((img: any) => ({
          ...img,
          sourceType: img.sourceType?.toLowerCase() ?? 'generated',
          metadata: {
            ...img.metadata,
            fileSize: img.metadata?.fileSize ?? 0,
            mimeType: img.metadata?.mimeType ?? 'image/png',
          },
        })),
      };
    },
  },
];

function migrateProject(data: any): Project {
  let currentVersion = data.version || '1.0.0';
  let migratedData = data;

  for (const migration of migrations) {
    if (migration.fromVersion === currentVersion) {
      migratedData = migration.migrate(migratedData);
      currentVersion = migration.toVersion;
    }
  }

  return migratedData as Project;
}
```
