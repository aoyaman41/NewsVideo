### 3.6.2 エラーハンドリングフロー

```typescript
// src/utils/errorHandler.ts

class AppError extends Error {
  constructor(
    public code: ErrorCode,
    message: string,
    public recoverable: boolean = true,
    public details?: any
  ) {
    super(message);
  }
}

/**
 * リトライ可能なエラーかどうかを判定
 */
function isRetryableError(error: unknown): boolean {
  if (error instanceof AxiosError) {
    const status = error.response?.status;
    // 429 (Rate Limit) と一部の5xx エラーのみリトライ
    return status === 429 || status === 500 || status === 502 || status === 503 || status === 504;
  }
  // ネットワークエラー（接続失敗など）もリトライ対象
  if (error instanceof Error && error.message.includes('ECONNREFUSED')) {
    return true;
  }
  return false;
}

/**
 * ジッター付き指数バックオフでリトライするAPI呼び出しラッパー
 */
async function handleApiCall<T>(
  apiCall: () => Promise<T>,
  options: {
    maxRetries?: number;
    baseDelayMs?: number;
    maxDelayMs?: number;
    maxTotalTimeMs?: number;
    onRetry?: (attempt: number, delayMs: number, error: Error) => void;
  } = {}
): Promise<T> {
  const {
    maxRetries = 3,
    baseDelayMs = 1000,
    maxDelayMs = 30000,
    maxTotalTimeMs = 120000,  // 合計最大2分
    onRetry,
  } = options;

  const startTime = Date.now();
  let lastError: Error | null = null;

  for (let attempt = 1; attempt <= maxRetries; attempt++) {
    try {
      return await apiCall();
    } catch (error) {
      lastError = error as Error;

      // リトライ不可のエラーは即座に終了
      if (!isRetryableError(error)) {
        throw error;
      }

      // 最大リトライ回数に達した
      if (attempt === maxRetries) {
        throw error;
      }

      // 指数バックオフ + ジッター計算
      // delay = min(maxDelay, baseDelay * 2^(attempt-1)) * (0.5 + random * 0.5)
      const exponentialDelay = Math.min(maxDelayMs, baseDelayMs * Math.pow(2, attempt - 1));
      const jitter = 0.5 + Math.random() * 0.5;
      const delayMs = Math.floor(exponentialDelay * jitter);

      // 合計時間チェック
      if (Date.now() - startTime + delayMs > maxTotalTimeMs) {
        throw new Error(`Max total retry time (${maxTotalTimeMs}ms) exceeded`);
      }

      onRetry?.(attempt, delayMs, lastError);
      await sleep(delayMs);
    }
  }

  throw lastError ?? new Error('Unreachable');
}

function sleep(ms: number): Promise<void> {
  return new Promise((resolve) => setTimeout(resolve, ms));
}

// 使用例
async function generateScript(article: Article): Promise<Part[]> {
  return handleApiCall(
    () => openaiClient.generateScript(article),
    {
      maxRetries: 3,
      baseDelayMs: 2000,
      onRetry: (attempt, delayMs, error) => {
        console.log(`Retry ${attempt} after ${delayMs}ms: ${error.message}`);
        // UIに進捗表示
        jobQueue.updateStatus({ retrying: true, attempt, nextRetryMs: delayMs });
      },
    }
  );
}
```

### 3.10.2 ジョブキューサービス

```typescript
// electron/services/jobQueue.ts

interface Job {
  id: string;
  type: JobType;
  status: JobStatus;
  payload: any;
  priority: number;
  createdAt: Date;
  startedAt?: Date;
  completedAt?: Date;
  error?: string;
  retryCount: number;
  maxRetries: number;
}

type JobType =
  | 'script_generation'
  | 'image_prompt_generation'
  | 'image_generation'
  | 'audio_generation'
  | 'video_render';

type JobStatus =
  | 'pending'
  | 'running'
  | 'completed'
  | 'failed'
  | 'cancelled';

interface JobQueueConfig {
  concurrency: {
    script_generation: 1;
    image_prompt_generation: 1;
    image_generation: 2;
    audio_generation: 3;
    video_render: 1;
  };
  retryPolicy: {
    maxRetries: 3;
    baseDelayMs: 1000;
    maxDelayMs: 30000;
    backoffMultiplier: 2;
  };
  circuitBreaker: {
    failureThreshold: 5;
    resetTimeoutMs: 30000;
  };
}

class JobQueueService {
  private queues: Map<JobType, Job[]> = new Map();
  private running: Map<JobType, Set<string>> = new Map();
  private circuitState: Map<JobType, CircuitState> = new Map();

  async enqueue(job: Omit<Job, 'id' | 'status' | 'createdAt' | 'retryCount'>): Promise<string> {
    const id = crypto.randomUUID();
    const newJob: Job = {
      ...job,
      id,
      status: 'pending',
      createdAt: new Date().toISOString(),
      retryCount: 0,
    };

    this.getQueue(job.type).push(newJob);
    this.processQueue(job.type);
    return id;
  }

  async cancel(jobId: string): Promise<boolean> {
    // キャンセル処理
  }

  async getStatus(jobId: string): Promise<Job | null> {
    // ステータス取得
  }

  private async processQueue(type: JobType): Promise<void> {
    const config = this.config.concurrency[type];
    const running = this.running.get(type) || new Set();
    const queue = this.getQueue(type);

    while (running.size < config && queue.length > 0) {
      const job = queue.shift()!;
      running.add(job.id);
      this.executeJob(job).finally(() => {
        running.delete(job.id);
        this.processQueue(type);
      });
    }
  }

  private async executeJob(job: Job): Promise<void> {
    // サーキットブレーカーチェック
    if (this.isCircuitOpen(job.type)) {
      await this.waitForCircuitReset(job.type);
    }

    job.status = 'running';
    job.startedAt = new Date();
    this.emitProgress(job);

    try {
      await this.runJobWithRetry(job);
      job.status = 'completed';
      job.completedAt = new Date();
      this.recordSuccess(job.type);
    } catch (error) {
      job.status = 'failed';
      job.error = error.message;
      this.recordFailure(job.type);
    }

    this.emitProgress(job);
  }

  private async runJobWithRetry(job: Job): Promise<void> {
    const { maxRetries, baseDelayMs, backoffMultiplier } = this.config.retryPolicy;

    while (job.retryCount <= maxRetries) {
      try {
        await this.executeJobHandler(job);
        return;
      } catch (error) {
        if (!this.isRetryableError(error) || job.retryCount >= maxRetries) {
          throw error;
        }
        job.retryCount++;
        const delay = Math.min(
          baseDelayMs * Math.pow(backoffMultiplier, job.retryCount - 1),
          this.config.retryPolicy.maxDelayMs
        );
        await this.sleep(delay);
      }
    }
  }

  private emitProgress(job: Job): void {
    // IPC経由でrendererに進捗を通知
    // ※チャンネル名は 'progress:update' に統一（ホワイトリスト許可済み）
    this.mainWindow.webContents.send('progress:update', {
      source: 'job',           // 進捗元の識別
      jobId: job.id,
      type: job.type,
      status: job.status,
      error: job.error,
    });
  }
}
```

```typescript
class CircuitBreaker {
  private state: CircuitState = { state: 'closed', failureCount: 0 };

  recordSuccess(): void {
    this.state.failureCount = 0;
    this.state.state = 'closed';
  }

  recordFailure(): void {
    this.state.failureCount++;
    this.state.lastFailureTime = new Date();

    if (this.state.failureCount >= this.config.failureThreshold) {
      this.state.state = 'open';
      this.state.nextRetryTime = new Date(
        Date.now() + this.config.resetTimeoutMs
      );
    }
  }

  isOpen(): boolean {
    if (this.state.state === 'open') {
      if (new Date() >= this.state.nextRetryTime!) {
        this.state.state = 'half-open';
        return false;
      }
      return true;
    }
    return false;
  }
}
```

```typescript
// electron/services/ffmpegWrapper.ts
import ffmpeg from 'fluent-ffmpeg';
import ffmpegStatic from 'ffmpeg-static';
import ffprobeStatic from 'ffprobe-static';
import { app } from 'electron';
import * as path from 'path';

// ffmpegのパス設定
const ffmpegPath = app.isPackaged
  ? path.join(process.resourcesPath, 'ffmpeg', 'ffmpeg')
  : ffmpegStatic;

const ffprobePath = app.isPackaged
  ? path.join(process.resourcesPath, 'ffmpeg', 'ffprobe')
  : ffprobeStatic.path;

ffmpeg.setFfmpegPath(ffmpegPath);
ffmpeg.setFfprobePath(ffprobePath);

export class FFmpegWrapper {
  /**
   * 音声ファイルの長さを取得
   */
  async getAudioDuration(filePath: string): Promise<number> {
    return new Promise((resolve, reject) => {
      ffmpeg.ffprobe(filePath, (err, metadata) => {
        if (err) {
          // メタデータ取得失敗時のフォールバック
          // ファイルサイズと推定ビットレートから計算
          resolve(this.estimateDuration(filePath));
          return;
        }
        resolve(metadata.format.duration || 0);
      });
    });
  }

  /**
   * パート動画を生成（画像+音声）
   * ※静止画の時間制御: -loop 1 と -t で各画像の表示時間を指定し、
   *   filter_complex の concat で連結
   */
  async createPartVideo(
    images: Array<{ path: string; duration: number }>,
    audioPath: string,
    outputPath: string,
    options: { fps: number; resolution: string; useHwAccel: boolean }
  ): Promise<void> {
    return new Promise((resolve, reject) => {
      let command = ffmpeg();
      const n = images.length;

      // 各画像を入力として追加（-loop 1 と -t で表示時間を制御）
      images.forEach((img) => {
        command = command
          .input(img.path)
          .inputOptions(['-loop', '1', '-t', img.duration.toFixed(3)]);
      });

      // 音声を追加
      command = command.input(audioPath);

      // filter_complex で画像をスケール統一・連結
      const filterInputs = Array.from({ length: n }, (_, i) => `[${i}:v]`);
      const scaleFilters = filterInputs.map(
        (input, i) => `${input}scale=${options.resolution.replace('x', ':')}:force_original_aspect_ratio=decrease,pad=${options.resolution.replace('x', ':')}:(ow-iw)/2:(oh-ih)/2,setsar=1[v${i}]`
      );
      const concatInputs = Array.from({ length: n }, (_, i) => `[v${i}]`).join('');
      const filterComplex = [
        ...scaleFilters,
        `${concatInputs}concat=n=${n}:v=1:a=0[vout]`
      ].join(';');

      command = command.complexFilter(filterComplex, ['vout']);

      // 音声マッピング（最後の入力が音声）
      command = command.outputOptions([`-map`, `[vout]`, `-map`, `${n}:a`]);

      // VideoToolboxハードウェアアクセラレーション
      if (options.useHwAccel) {
        command = command
          .outputOptions([
            '-c:v', 'h264_videotoolbox',
            '-profile:v', 'high',
            '-level', '4.2',
          ]);
      } else {
        command = command
          .outputOptions([
            '-c:v', 'libx264',
            '-preset', 'medium',
            '-crf', '23',
          ]);
      }

      command
        .outputOptions([
          `-r`, `${options.fps}`,
          '-c:a', 'aac',
          '-b:a', '192k',
          '-pix_fmt', 'yuv420p',
          '-shortest',           // 音声長に合わせて終了
          '-movflags', '+faststart',  // Web再生最適化（メタデータを先頭に配置）
        ])
        .output(outputPath)
        .on('progress', (progress) => {
          // 進捗通知
        })
        .on('end', () => resolve())
        .on('error', (err) => reject(err))
        .run();
    });
  }

  /**
   * 動画を連結
   */
  async concatenateVideos(
    inputPaths: string[],
    outputPath: string
  ): Promise<void> {
    // concat demuxerを使用
    const listFile = await this.createConcatListFile(inputPaths);

    return new Promise((resolve, reject) => {
      ffmpeg()
        .input(listFile)
        .inputOptions(['-f', 'concat', '-safe', '0'])
        .outputOptions(['-c', 'copy', '-movflags', '+faststart'])  // Web再生最適化
        .output(outputPath)
        .on('end', () => {
          fs.unlinkSync(listFile);
          resolve();
        })
        .on('error', (err) => {
          fs.unlinkSync(listFile);
          reject(err);
        })
        .run();
    });
  }

  private estimateDuration(filePath: string): number {
    // MP3の場合、ファイルサイズと128kbpsを仮定して推定
    const stats = fs.statSync(filePath);
    const bitrateKbps = 128;
    return (stats.size * 8) / (bitrateKbps * 1000);
  }
}
```

### 3.12.2 画像切り替えタイミング計算

```typescript
/**
 * パート内の画像切り替えタイミングを計算
 * デフォルト: 音声長 / 画像枚数 で均等割り、端数は最後の画像に吸収
 */
function calculateImageTimings(
  audioDurationSec: number,
  imageCount: number
): Array<{ startSec: number; endSec: number }> {
  if (imageCount === 0) return [];
  if (imageCount === 1) {
    return [{ startSec: 0, endSec: audioDurationSec }];
  }

  const interval = audioDurationSec / imageCount;
  const timings: Array<{ startSec: number; endSec: number }> = [];

  for (let i = 0; i < imageCount; i++) {
    const startSec = interval * i;
    const endSec = i === imageCount - 1
      ? audioDurationSec  // 最後の画像は端数を吸収
      : interval * (i + 1);

    timings.push({
      startSec: Math.round(startSec * 1000) / 1000,
      endSec: Math.round(endSec * 1000) / 1000
    });
  }

  return timings;
}
```
