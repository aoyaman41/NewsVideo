```
## 要件
1. 記事を論理的な単位（パート）に分割してください
2. 各パートは30秒〜60秒程度で読み上げられる長さにしてください（150〜300文字程度）

## 出力形式
JSON形式で以下の構造で出力してください：
{
  "parts": [
    {
      "title": "パートタイトル",
      "summary": "パートの要約",
      "scriptText": "読み上げ用原稿テキスト",
      "estimatedDurationSec": 推定読み上げ時間（秒）
    }
  ]
}
```

```typescript
// src/services/ai/openaiClient.ts

import { randomUUID } from 'node:crypto';
import axios from 'axios';

const OPENAI_BASE_URL = 'https://api.openai.com/v1';

export class OpenAIClient {
  private apiKey: string;

  constructor(apiKey: string) {
    this.apiKey = apiKey;
  }

  async generateScript(article: Article, options: ScriptOptions): Promise<Part[]> {
    const response = await axios.post(
      `${OPENAI_BASE_URL}/chat/completions`,
      {
        model: 'gpt-5.2',
        messages: [
          { role: 'system', content: SCRIPT_GENERATION_PROMPT },
          { role: 'user', content: this.formatArticleInput(article) },
        ],
        response_format: { type: 'json_object' },
        reasoning_effort: 'medium',
        temperature: 0.7,
      },
      {
        headers: {
          'Authorization': `Bearer ${this.apiKey}`,
          'Content-Type': 'application/json',
        },
        timeout: 60000, // 60秒
      }
    );

    const result = JSON.parse(response.data.choices[0].message.content);
    return this.mapToParts(result.parts);
  }

  private formatArticleInput(article: Article): string {
    return `
タイトル: ${article.title}
${article.source ? `出典: ${article.source}` : ''}

本文:
${article.bodyText}
    `.trim();
  }

  // ※統一規約に準拠: 必須フィールド完備、日時はISO文字列
  private mapToParts(rawParts: any[]): Part[] {
    const now = new Date().toISOString();
    return rawParts.map((p, index) => ({
      id: randomUUID(),
      index,
      title: p.title,
      summary: p.summary ?? '',
      scriptText: p.scriptText,
      durationEstimateSec: p.estimatedDurationSec ?? 0,
      panelImages: [],
      comments: [],
      createdAt: now,           // ← 追加
      updatedAt: now,           // ← 追加
      scriptGeneratedAt: now,   // ← 追加
      scriptModifiedByUser: false,  // ← 追加
    }));
  }
}
```

### 4.5.2 Google AI Studio 画像生成クライアント

```typescript
// src/services/image/googleImagenClient.ts

import { randomUUID } from 'node:crypto';
import axios from 'axios';
import * as fs from 'fs/promises';
import * as path from 'path';

const GOOGLE_AI_BASE_URL = 'https://generativelanguage.googleapis.com/v1beta';

/**
 * Google AI Studio 画像生成クライアント
 * コード内識別子: google_imagen
 */
export class GoogleImagenClient {
  private apiKey: string;

  constructor(apiKey: string) {
    this.apiKey = apiKey;
  }

  async generateImage(prompt: ImagePrompt, outputPath: string): Promise<ImageAsset> {
    const response = await axios.post(
      `${GOOGLE_AI_BASE_URL}/models/gemini-3-pro-image:generateContent`,
      {
        contents: [{
          parts: [{ text: this.buildPromptText(prompt) }],
        }],
        generationConfig: {
          responseModalities: ['IMAGE'],
          imageDimensions: this.getImageDimensions(prompt.aspectRatio),
        },
      },
      {
        params: { key: this.apiKey },
        headers: { 'Content-Type': 'application/json' },
        timeout: 120000, // 120秒
      }
    );

    const inlineData = response.data.candidates[0].content.parts[0].inlineData;
    const mimeType = inlineData?.mimeType ?? 'image/png';
    const buffer = Buffer.from(inlineData.data, 'base64');

    await fs.writeFile(outputPath, buffer);

    const dimensions = this.getImageDimensions(prompt.aspectRatio);

    // ※統一規約に準拠: sourceType小文字、日時はISO文字列、metadata完全版
    return {
      id: randomUUID(),
      filePath: outputPath,
      sourceType: 'generated',  // ← 小文字に統一
      metadata: {
        width: dimensions.width,
        height: dimensions.height,
        mimeType,                               // ← 追加
        fileSize: buffer.byteLength,            // ← 追加
        createdAt: new Date().toISOString(),    // ← ISO文字列に統一
        promptId: prompt.id,
        tags: [],
      },
    };
  }

  private buildPromptText(prompt: ImagePrompt): string {
    let text = prompt.prompt;
    // stylePresetを使用（styleは非推奨）
    if (prompt.stylePreset) {
      text = `${prompt.stylePreset} style. ${text}`;
    }
    if (prompt.negativePrompt) {
      text += ` Avoid: ${prompt.negativePrompt}`;
    }
    return text;
  }

  private getImageDimensions(aspectRatio: string): { width: number; height: number } {
    switch (aspectRatio) {
      case '16:9': return { width: 1920, height: 1080 };
      case '9:16': return { width: 1080, height: 1920 };
      case '1:1': return { width: 1080, height: 1080 };
      default: return { width: 1920, height: 1080 };
    }
  }
}
```

```typescript
// src/services/tts/googleTTSClient.ts

import { randomUUID } from 'node:crypto';
import { execFile } from 'node:child_process';
import { promisify } from 'node:util';
import axios from 'axios';
import * as fs from 'fs/promises';
import ffprobeStatic from 'ffprobe-static';

const execFileAsync = promisify(execFile);
const TTS_BASE_URL = 'https://texttospeech.googleapis.com/v1';

export class GoogleTTSClient {
  private apiKey: string;

  constructor(apiKey: string) {
    this.apiKey = apiKey;
  }

  async synthesize(text: string, options: TTSOptions, outputPath: string): Promise<AudioAsset> {
    const response = await axios.post(
      `${TTS_BASE_URL}/text:synthesize`,
      {
        input: { text },
        voice: {
          languageCode: options.languageCode,
          name: options.voiceName,
        },
        audioConfig: {
          audioEncoding: options.audioEncoding,
          speakingRate: options.speakingRate,
          pitch: options.pitch,
        },
      },
      {
        params: { key: this.apiKey },
        headers: { 'Content-Type': 'application/json' },
        timeout: 30000,
      }
    );

    const audioBuffer = Buffer.from(response.data.audioContent, 'base64');
    await fs.writeFile(outputPath, audioBuffer);

    // ffprobeで実測の音声長を取得
    const durationSec = await this.getAudioDuration(outputPath);

    // ※統一規約に準拠: AudioAsset完全版
    return {
      id: randomUUID(),
      filePath: outputPath,
      durationSec,
      ttsEngine: 'google_tts',       // ← 追加（必須フィールド）
      voiceId: options.voiceName,    // ← voiceProfile → voiceId に変更
      settings: {
        speakingRate: options.speakingRate,
        pitch: options.pitch,
        languageCode: options.languageCode,
      },
      generatedAt: new Date().toISOString(),  // ← 追加
    };
  }

  async getVoices(languageCode?: string): Promise<VoiceInfo[]> {
    const response = await axios.get(`${TTS_BASE_URL}/voices`, {
      params: {
        key: this.apiKey,
        languageCode,
      },
    });

    return response.data.voices.map((v: any) => ({
      name: v.name,
      languageCodes: v.languageCodes,
      gender: v.ssmlGender,
      sampleRateHertz: v.naturalSampleRateHertz,
    }));
  }

  /**
   * ffprobeを使用して音声ファイルの実測長を取得
   * タイムライン表示の正確性を担保
   */
  private async getAudioDuration(filePath: string): Promise<number> {
    try {
      const { stdout } = await execFileAsync(ffprobeStatic.path, [
        '-v', 'error',
        '-show_entries', 'format=duration',
        '-of', 'default=noprint_wrappers=1:nokey=1',
        filePath
      ]);
      const sec = parseFloat(stdout.trim());
      return Number.isFinite(sec) ? sec : 0;
    } catch (error) {
      console.error('Failed to get audio duration:', error);
      return 0;
    }
  }
}
```
